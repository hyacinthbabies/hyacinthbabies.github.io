一、Object类型

创建Object实例的方式2种：
  • 使用new后跟Object构造函数
var person = new Object();
person.name = "shq";
  • 使用对象字面量表示法
var person = {
   name : "shq"
};
访问属性的方式2中： 点表示法和 方括号表示
alert(person.name);//点表示法（最好使用此）
alert(person["name"]);//方括号表示法
var propertyName = "name";
alert(person[propertyName]);//方括号访问变量

二、Array类型

创建数组的基本方式2种：
  • 使用new后跟Array构造函数，new可省略
var colors = new Array();//new 
var colors = new Array(20);//数组长度为20
var colors = new Array("red","blue","green");//包含3个字符串的数组
  • 使用数组字面量表示法
var colors = ["red","blue","green"];
访问值的方式：
alert(colors[0]);//red，索引从0-(length-1)
alert(colors.length);//长度，3

1.检测数组

对于一个网页或者一个全局作用域： instanceof
ECMAScript5： Array.isArray()

2.转换方法

返回由数组中每个值的字符串形式拼接而成的一个以逗号分隔的字符串 toLocaleString()、toString()、valueOf()、join()
var colors = ["red","blue","green"];
alert(colors.toString());//red,blue,green
alert(colors.toLocaleString());//red,blue,green
alert(colors.valueOf());//red,blue,green
alert(colors);//red,blue,green 会在后台调用toString()方法
alert(colors.join(","));//red,blue,green
alert(colors.join("||"));//red||blue||green

3.栈方法

数组可以表现得像栈一样，LIFO 后进先出
插入： push()逐个添加到数字末尾，并返回修改后数组的长度
移除： pop()从数组末尾移除最后一项，返回移除的项
var colors = new Array();
var count = colors.push("red","green");
alert(count);//2
alert(colors);//["red","green"]
var item = colors.pop();
alert(item);//"green"

4.队列方法

数组也可以表现得像队列一样，FIFO 先进先出
插入： unshift()在数组前端添加任意个项并返回新数组的长度
移除： shift()移除数组中的第一个项并返回该项
var colors = new Array();
var count = colors.unshift("red","green");
alert(count);//2
alert(colors);//["green","red"]
var item = colors.shift();
alert(item);//"green" 
push()+shift()实现先进先出、pop()+unshift()实现后进后出

5.重排序方法

reverse()反转数组项的顺序
sort()默认升序排列，会调用toString()进行转型后比较字符串
sort()可以接收函数作为参数function(arg1,arg2)
var values = [0,1,5,10,15];
values.sort();
alert(values);//0,1,10,15,5
//利用函数纠正排序
function compare(value1,value2){
  if(value1 < value2){
      return -1;
   }else if(value1 > value2){
       return 1;
   }else{
       return 0;
   }
   //对于数值类型或valueOf()方法返回数值类型的对象类型,直接使用
   //return value2-value1;//升序
}
var values = [0,1,5,10,15];
values.sort(compare);
alert(values);//0,1,5,10,15

6.操作方法

concat()创建当前数值的一个副本，然后接收到的参数添加到这个副本的末尾，构建新数值
var colors = ["red","green","blue"];
var colors2 = colors.concat("yellow",["black","brown"]);
alert(colors);//red,green,blue
alert(colors2);//red,green,blue,yellow,black,brown
slice()能够基于当前数组中的一个或多个项创建一个数组
接收一或两个参数，即 slice(返回项的起始,结束位置)
var colors = ["red","green","blue","yellow","purple"];
var colors2 = colors.slice(1);
var colors3 = colors.slice(1,4);
var colors4 = colors.slice(-1);//-1+colors.length = 4
alert(colors2);//green,blue,yellow,purplr
alert(colors3);//green,blue,yellow
alert(colors4);//purple
splice()主要用途是向数组的中部插入项，使用方法有3种
  • 删除: splice(起始位置，要删除的项数)
var colors = ["red","green","blue"];
var removed = splice(0,1);//删除第一项
alert(colors);//green,blue
alert(removed);//red
  • 插入: splice(起始位置，0，要插入的项)
var colors = ["red","green","blue"];
var removed = splice(1,0,"yellow","orange");//从位置1开始插入2项
alert(colors);//red,green,yellow,orange,blue
alert(removed);//返回的是一个空数组
  • 替换: splice(起始位置，要删除的项数，要插入的任意数量的项)
var colors = ["red","green","blue"];
var removed = splice(1,1,"yellow","purple");//插入2项，删除1项
alert(colors);//red,yellow,purple,blue
alert(removed);//green,返回的数组中只包含一项

7.位置方法

返回要查找的项在数组中的位置，或在没找到的情况下返回-1(查找必须严格相等， ===)
indexOf():从数组的开头(0)开始向后找
indexOf(要查找的项,表示查找起点位置的索引(可选))
var numbers = [1,2,3,4,5,4,3,2,1];
alert(numbers.indexOf(4));//3
alert(numbers.indexOf(4,4));//5
var person = {name:"Nic"};
var people = [{name:"Nic"}];
var morePeople = [person];
alert(people.indexOf(person));//-1
alert(morePeople.indexOf(person));//0
lastIndexOf():从数组的末尾开始向前找
lastIndexOf(要查找的项,表示查找起点位置的索引(可选))
var numbers = [1,2,3,4,5,4,3,2,1];
alert(numbers.lastIndexOf(4));//5
alert(numbers.lastIndexOf(4,4));//3
浏览器：IE9+、Firefox2+、Safari3+、Opera9.5、Chrome

8.迭代方法

迭代方法(要在每一项上运行的函数，运行该函数的作用域对象--影响this的值(可选))
传入的函数(数组项的值,该项在数组中的位置,数组对象本身)
  • every():对数组中的每一项运行给定函数，如果该函数对每一项都返回true，则返回true
var numbers = [1,2,3,4,5,4,3,2,1];
var everyResult = numbers.every(function(item,index,array){
  return (item >2);
});
alert(everyResult);//false
  • filter():对数组中的每一项运行给定函数，返回该函数会返回true的项组成的数组
var numbers = [1,2,3,4,5,4,3,2,1];
var filterResult = numbers.filter(function(item,index,array){
  return (item >2);
});
alert(filterResult);//[3,4,5,4,3]
  • forEach():对数组中的每一项运行给定函数，这个方法没有返回值
//本质上与for循环迭代数组一样
var numbers = [1,2,3,4,5,4,3,2,1];
numbers.forEach(function(item,index,array){
  //执行某些操作
});
  • map():对数组中的每一项运行给定函数，返回每次函数调用的结果组成的数组
var numbers = [1,2,3,4,5,4,3,2,1];
var mapResult = numbers.map(function(item,index,array){
  return item * 2
});
alert(mapResult);//[2,4,6,8,10,8,6,4,2]
  • some():对函数中的每一项运行给定函数，如果该函数对任一项返回true，则返回true
var numbers = [1,2,3,4,5,4,3,2,1];
var someResult = numbers.some(function(item,index,array){
  return (item >2);
});
alert(someResult);//true
浏览器：IE9+、Firefox2+、Safari3+、Opera9.5、Chrome

9.缩小方法

迭代数组的所有项，然后构建一个最终返回的值
缩小方法(在每一项上调用的函数,作为缩小基础的初始值(可选))
传入的函数(前一个值，当前值，项的索引，数组对象)返回的任何值作为第一个参数自动传给下一项
reduce():从数组的第一项开始，逐个遍历到最后。执行求数组中的所有值之和的操作
var values = [1,2,3,4,5];
var sum = values.reduce(function(prev,cur,index,array){
  return prev + cur;//第一次执行prev是1，第二次执行prev是3（1+2）
});
alert(sum);//15
reduceRight():从数组的最后一项开始，向前遍历到第一项
var values = [1,2,3,4,5];
var sum = values.reduceRight(function(prev,cur,index,array){
  return prev + cur;//第一次执行prev是5，第二次执行prev是9（5+4）
});
alert(sum);//15
浏览器：IE9+、Firefox3+、Safari4+、Opera10.5、Chrome

Date类型使用自UTC国际协调时间1970年1月1日午夜(零时)开始经过的毫秒数来保存日期
创建日期对象，使用new操作符和Date构造函数即可
var now = new Date();//获取当前日期和时间
alert(now);//Wed Dec 21 2016 11:08:11 GMT+0800 (中国标准时间)
根据特定日期和时间创建日期对象:
Date.parse():接收表示日期的字符串参数，返回相应日期的毫秒数。如果不能表示日期，则返回NaN
var someDate = new Date(Date.parse("May 25,2004"));
var someDate = new Date("May 25,2004");//与调用parse等价，不用浏览器实现可能不同
alert(someDate);//Tue May 25 2004 00:00:00 GMT+0800 (中国标准时间)
Date.UTC():返回表示日期的毫秒数
date.UTC(年份，基于0的月份0-11，月中哪一天1-31(默认1），小时数0-23(默认0)，分钟(默认0)，秒(默认0)，毫秒数(默认0))只有前2个参数必须
//有疑问
//按理是GMT时间2005年5月5日下午5:55:55
var all = new Date(Date.UTC(2005,4,5,17,55,55));
alert(all);//Fri May 06 2005 01:55:55 GMT+0800 (中国标准时间)
//本地时间2005年5月5日下午5:55:55

Date().now():返回表示调用这个方法时的日期和时间的毫秒数
Date.now();//1482300835521

四、RegExp类型

创建正则表达式：1字面量形式，2RegExp构造函数
  • 字面量形式：
var expression = / pattern / flags ;
pattern模式：可以包含字符类、限定符、分组、向前查找以及反向引用
flags:可以有一个或多个标志，用以标明正则表达式的行为
正则表达式的匹配模式支持下列3个标志：
g:表示全局global模式，即模式将被应用于所有字符串，而非在发现第一个匹配项时立即停止
i:表示不区分大小写模式，即在确定匹配项时忽略模式与字符串的大小写
m:表示多行模式，即在到达一行文本末尾时还会继续查找下一行中是否存在与模式匹配的项
一个正则表达式是一个模式与上述3个标志的组合体
//匹配字符串中所有“at”的实例
var pattern1 = /at/g;
//匹配第一个“bat”或“cat”，不区分大小写
var pattern2 = /[bc]at/i;
//匹配多有以“at”结尾的3个字符串的组合，不区分大小写
var pattern3 = /.at/gi;
元字符都必须转义，元字符包括：
（ [ { \ ^ $ | } ? * + . ] ）
//匹配第一个“[bc]at”,不区分大小写
var pattern1 = /\[bc\]at/i;
//匹配所有“.at”,不区分大小写
var pattern2 = /\.at/gi;
  • 构造函数形式： RegExp(匹配的字符串模式，可选的标志字符串)
//匹配第一个“bat”或“cat”，不区分大小写
var pattern1 = new RegExp("[bc]at","i");
由于构造函数模式参数是字符串，所以某些情况字符需要双重转义：
字面量模式 等价的字符串
/\[bc\]at/ "\\[bc\\]at/"
/\.at/ "\\.at"
/name\/age/ "name\\/age"
/\d.\d{1,2}/ "\\d.\\d{1,2}"
/\w\\hello\\123/ "\\w\\\\hello\\\\123"

ECMAScript5规定：使用正则表达式字面量和使用regExp构造函数一样，每次都创建新的RegExp实例

1.RegExp实例属性（意义不大）

  • global：布尔值，表示是否设置了g标志
  • ignoreCase：布尔值，表示是否设置了i标志
  • lastIndex：整数，表示开始搜索下一个匹配项的字符位置，从0算起
  • multiline：布尔值，表示是否设置了m标志
  • source：正则表达式的字符串表示，安卓字面量形式而非传入构造函数中的字符串模式返回

2.RegExp实例方法

exec():专门捕获组而设计的
exec(要应用模式的字符串),返回包含第一个匹配项信息的数组（包含2个属性：index和input）；或者没有匹配项的情况返回null
index：表示匹配项在字符串中的位置
input：表示应用正则表达式的字符串
var text = "mom and dad and baby";
var pattern = /mom( and dad( and baby)?)?/gi;
var matches = pattern.exec(text);
alert(matches.index);//0
alert(matches.input);//"mom and dad and baby"
//第一项是匹配的整个字符串
alert(matches[0]);//"mom and dad and baby"
//第二项包含与第一个捕获组匹配的内容
alert(matches[1]);//" and dad and baby"
//第三项包含与第二个捕获组匹配的内容
alert(matches[2]);//" and baby"
非全局模式下：每次调用exec()返回的都是第一个匹配项，lastIndex的值始终保持不变。全局模式下：每次调用exec()返回的都是下一个匹配项，lastIndex的值增加。
test()：接收一个参数，模式与参数匹配的情况返回true，否则返回false
var text = "000-00-0000";
var pattern = /\d{3}-\d{2}-\d{4}/;
if(pattern.test(text)){
  alert("匹配");
}
RegExp实例继承的 toLocaleString()和toString()都会返回正则表达式的字面量，与创建正则表达式方式无关。正则表达式 valueOf返回正则表达式本身

3.RegExp构造函数属性

这些属性分别有一个长属性名和一个短属性名（Opera除外，不支持短属性名）
长属性名 短属性名 说明
input $_ 最近一次要匹配的字符串。Opera不支持此属性
lastMatch $& 最近一次的匹配项。Opera不支持此属性
lastParen $+ 最近一次匹配的捕获组。Opera不支持此属性
leftContext $` input字符串中lastMatch之前的文本
multiline $* 布尔值，表示是否所有表达式都使用多行模式。ie、Opera不支持此属性
rightContext $' input字符串中lastMatch之后的文本
还有另外9个属性，访问RegExp.$1...RegExp.$9，分别用于存储第一...第九个撇皮的捕获组，在调用exec和test时会被自动填充
var text = "this has been a short summer";
var pattern = /(.)hort/g;
if(pattern.test(text)){
 alert(RegExp.input);// this has been a short summer
 alert(RegExp.leftContext);// this has been a
 alert(RegExp.["$`"]);
 alert(RegExp.rightContext);// summer
 alert(RegExp.lastMatch);// short
 alert(RegExp.lastParen);// s
 alert(RegExp.multiline);// false
}
var text = "this has been a short summer";
var pattern = /(..)or(.)/g;
if(pattern.test(text)){
 alert(RegExp.$1);// sh
 alert(RegExp.$2);// t
}

4.函数内部属性

函数内部，2个特殊的对象： argumnents和this，
argumnents：是一个类数组对象，包含传入函数中的所有参数。它有一个callee的属性，它是一个指针，指向拥有这个arguments对象的函数
function factorial(num){
  if(num <= 1){
      return 1;
   }else{
      //return num * factorial(num-1)
      return num * arguments.callee(num-1)
   }
}
this：引用的是函数可以执行的环境对象，也可以说this值，当在全局作用域调用函数，this指代window。
window.color = "red";
var o = {color:"blue"};
function sayColor(){
  alert(this.color);
}
sayColor();//red
o.sayColor = sayColor;
o.sayColor();//blue
caller：ECMAScript5规范化另一个函数对象的属性，它保存着调用当前的函数的引用
function outer(){
   inner();
}
function inner(){
   alert(inner.caller);
   //alert(arguments.callee.caller);

四、跨源资源共享

基本思想：使用自定义HTTP头部让浏览器与服务器沟通，从而决定请求成功还是失败

1.IE对CORS的实现

概要：ie8引入了XDR对象，与XHR类似，但它能够实现安全可靠的跨域通信

XDR与XHR的区别:

  • cookies不会随请求发送，也不会随响应返回
  • 只能设置请求头部信息中的Content-Type字段
  • 不能访问响应头部信息
  • 只支持GET和POST请求

方法：
open(arg1,arg2)

参数：
arg1:请求的类型
arg2:url

XDR请求是异步执行，请求返回会触发load事件，响应的数据会保存在responseText中,如下所示： 

var xdr = new XDomainRequest();
xdr.onload = function(){
    alert(xdr.responseText);
};
//无法确定状态码，失败会触发error事件
xdr.onerror = function(){
    alert("An error occurred.");
};
xdr.timeout = 1000;
xdr.ontimeout = function(){
    alert("Request took too long");
};
xdr.open("post","http://www.somewhere-else.com/page/*);
//为支持POST请求，XDR对象提供contentType属性，用来表示数据发送格式
//此属性是XDR对象影响头部信息的唯一方式
xdr.contentType = "application/x-www-form-urlencoded";
xdr.send("name1=value1&name2=value2");

2.其他浏览器对CORS的实现

概要：Firefox3.5+、Safari4+、Chrome、IOS版Safari和Android平台的Webkit实现了对CORS的原生支持

通过跨域XHR对象可以访问status和statusText属性，支持同步请求,跨域对象也有限制：

  • 不能使用setRequestHeader()设置自定义头部
  • 不能发送和接收cookie

3.Preflighted Requests(预请求)

场景：使用下列高级选项发送请求时，便会向服务器发送Preflight请求：

  • Origin:与简单的请求相同
  • Access-Control-Request-Method：请求自身使用的方法
  • Access-Control-Request-Headers：(可选)自定义的头部信息，多个头部以逗号分隔
发送请求后，服务器通过在响应中发送如下头部与浏览器沟通：
  • Access-Control-Allow-Origin:与简单的请求相同
  • Access-Control-Allow-Methods：允许的方法，多个方法以逗号分隔
  • Access-Control-Allow-Headers：允许的头部，多个头部以逗号分隔
  • Access-Control-Max-Age：应该将这个Preflight请求缓存多长时间(以秒表示)
支持预请求的浏览器包括：Firefox3.5+、Safari4+、Chrome、ie11

4.带凭证的请求

凭据：cookie、HTTP认证及客户端SSL证明

通过将withCredentials属性设置为true，可以指定某个请求应发送凭据。

支持预请求的浏览器包括：Firefox3.5+、Safari4+、Chrome、ie11

function createCORSRequest(method,url){

3.Comet

概念：更高级的Ajax技术，有人称它服务器推送。它是一种服务器向页面推送数据的技术。

实现Comet的2种方式：长轮循和流

轮循：长轮循和短轮循，所有浏览器都支持轮循，使用XHR对象和setTimeout()就可以实现

HTTP流：关键所有服务器端语言都支持打印到输出缓存然后刷新(将输出缓存中的内容一次性全部发送到客户端)的功能
侦听readystatechange事件及检测readystate的值是否为3，利用XHR对象实现HTTP流。 

function createStreamingClient(url,progress,finished){
    var xhr = new XMLHttpRequest(),received = 0;
    xhr.open("get",url,true);
    xhr.onreadystatechange = function(){
        var result;
        if(xhr.readystate == 3){
           //只取得最新数据并调整计数器
           result = xhr.responseText.substring(received);
           received +=result.length;

           //调用progress回调函数
           progress(result);
        }else if(xhr.readystate == 4){
           finished(xhr.responseText);
        }
    };
    xhr.send(null);
    return xhr;
}
//createStreamingClient()接收三个参数：要连接的URL、在接收到数据时调用的函数以及关闭连接时调用的函数
var client = createStreamingClient("streaming.php",function(data){
       alert("Received:"+data);
    },function(data){
       alert("Done!")
    });

4.服务器发送事件

SSE:是围绕只读Comet交互推出的API或者模式。SSE API用于创建到服务器的单向连接，服务器通过这个连接可以发送任意数量的数据。服务器响应的MIME类型必须是text/event-stream

建立连接 

//浏览器向服务器发起连接,url服务器网址必须与当前网页的网址在同一个网域，协议端口必须相同
var source = new EventSource(url);
EventSource实例对象有一个readyState属性，表明连接所处的状态 source.readyState可以取3个值：
  • 0(EventSource.CONNECTING)，表示连接还未建立，或连接断线
  • 1(EventSource.OPEN),表示连接已经建立，可以接收数据
  • 2(EventSource.CLOSED),表示连接已断，且不会重连

open事件

//连接一旦建立，就会触发open事件，可以定义相应的回调函数
source.onopen = function(event){
    //handle open event
}

message事件

//收到数据就会触发message事件
source.onmessage = function(event) {
  var data = event.data;
  var origin = event.origin;
  var lastEventId = event.lastEventId;
  // handle message
};

参数对象event有如下属性：

  • data:服务器端传回的数据(文本格式)
  • origin:服务器端URL的域名部分，即协议、域名和端口
  • lastEventId:数据的编号，由服务器端发送。如果没有编号，这个属性为空

error事件

//如果发生通信错误（比如连接中断），就会触发error事件。
source.onerror = function(event) {
  // handle error event
};

自定义事件

//服务器可以与浏览器约定自定义事件。这种情况下，发送回来的数据不会触发message事件
source.addEventListener("foo", function(event) {
  var data = event.data;
  var origin = event.origin;
  var lastEventId = event.lastEventId;
  // handle message
}, false);
//上面表示浏览器对foo事件进行监听

close事件

//close方法用于关闭连接
source.close();

5.Web Sockets

概念：新浏览器API，目标是在一个单独的持久连接上提供全双工、双向通信。只有支持Web Socket协议的服务器才能工作。

Web Sockets使用自定义协议，URL模式也不同，未加密连接从http://-->ws://,加密的连接从https://-->wss://

创建Web Sockets

//实例一个webSocket对象并传入要连接的URL
var socket = new WebSocket("ws://www.example.com/server.php");
//发送数据(纯文本，其他格式需要序列化)
socket.send("Hello world!");
//服务器返回数据时(纯文本，需要解析)
socket.onmessage = function(event){
    var data = event.data;
    //处理数据
}
//关闭连接
socket.close()

表示当前状态的readyState属性：

  • WebSocket.OPENING(0):正在建立连接
  • WebSocket.OPEN(1):已经建立连接
  • WebSocket.CLOSING(2):正在关闭连接
  • WebSocket.CLOSE(3):已经关闭连接
其他事件(不支持DOM2级)：
  • open:在成功建立连接时触发
  • error：在发生错误时触发，连接不能连续
  • close:在连接关闭时触发。只有close事件event对象有其他三个属性：wasClean、code、reason
var socket = new WebSocket("ws://www.example.com/server.php");
socket.onopen = function(){//...};
socket.onerror = function(){//...};
socket.onclose = function(event){
    console.log(event.wasClean);//布尔值，表示连接是否已经关闭
    console.log(event.code);//服务器返回的数值状态码
    console.log(event.code);//字符串，包含服务器发回的消息
};
