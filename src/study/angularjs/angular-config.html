3.双向绑定 
    <!DOCTYPE html>
	<html lang="en">
	<head>
	    
		
	</head>
	<body ng-app="routeApp">
	    

	        ng-model绑定 HTML 元素 到应用程序数据
	        
	        {{testModel}}双向绑定，$scope.testModel值变{{testModel}}就会变
	    

	</body>
	</html>
    
        var myApp = angular.module("routeApp",[]);//ng-app,必须
        myApp.controller('RouteCtl', function($scope) {
            $scope.testModel = 'ng-model绑定 HTML 元素 到应用程序数据';
        });//ng-controller
