1.angular中$scope：最好不要把所有变量都绑在$scope上，属于同一块的，最好用一个对象，属性都写在对象中。

    $scope.obj={
        'code':'',//编号
        'name':'',//姓名
    }
    //调用
    $scope.obj.code = '123';
    $scope.obj.name = 'shq';

2.angular自身的路由已经提取出来，需要另外下载angular-route

   var routeApp = angular.module('routeApp',['ngRoute']);
   routeApp.config(['$routeProvider',function($routeProvider){
$routeProvider
.when('/column',{
    templateUrl:'column.html',
    css:'column.css'
})
.when('/table',{
    templateUrl:'table.html',
    controller:'RouteListCtl'
})
.otherwise({
     redirectTo:'/'
 });
}]);

2.为了更好使用单页面路由，使用angular-ui-router

  var myApp = angular.module("routeApp", ['ui.router']);
myApp.config(function($stateProvider, $urlRouterProvider) {
    $urlRouterProvider.when("", "/bootstrap/column");
    $stateProvider
        .state("bootstrap", {
            url: "/bootstrap",
            templateUrl: "/src/bootstrap/bootstrap.html",
            controller:'RouteListCtl',
            css:'/src/bootstrap/bootstrap.css'
        })
});
